def report(
    report_type: str = typer.Option(
        "all", "--type", "-t", help="Type de rapport √† g√©n√©rer (all, users-vms, status)"
    ),
    report_format: str = typer.Option(
        "all", "--format", "-f", help="Format de rapport (all, json, markdown, html)"
    ),

        "outputs", "--output-dir", "-o", help="R√©pertoire de sortie pour les rapports"

        typer.echo(f"‚ùå Format de rapport invalide: {report_format}")
        typer.echo("Formats valides: all, json, markdown, html")
        raise typer.Exit(1) from exc

    # Appeler directement la fonction
    generate_reports(report_type_enum, format_enum, output_dir, verbose)

def signup(
    name: str = typer.Option(
        "Jean Dupont", "--name", "-n", help="Nom de l'utilisateur"
    ),
    email: str = typer.Option(
        "jean@dupont21.com", "--email", "-e", help="Email de l'utilisateur"
    ),
    password: str = typer.Option(
        "password123", "--password", "-p", help="Mot de passe de l'utilisateur"
    ),
    verbose: bool = typer.Option(False, "--verbose", "-v", help="Mode verbeux"),
) -> None:
    """
    üë§ Cr√©er un nouvel utilisateur avec authentification

    Cr√©e un utilisateur via /auth/signup et r√©cup√®re son token d'authentification.

    Exemples:

    \b
    python main.py signup
    python main.py signup --name "Alice Martin" --email "alice@example.com"
    python main.py signup -n "Bob Dupont" -e "bob@test.com" -p "monmotdepasse" --verbose
    """
    from utils.api.auth import Auth
    from utils.config import config

    if verbose:
        typer.echo("üîß Configuration utilisateur:")
        typer.echo(f"   Nom: {name}")
        typer.echo(f"   Email: {email}")
        typer.echo(f"   Mot de passe: {'*' * len(password)}")
        typer.echo()

    logger.info("D√©but du processus de cr√©ation d'utilisateur", email=email, name=name)

    # Initialisation du client Auth
    auth = Auth(config.DEMO_API_BASE_URL)

    try:
        # Cr√©ation de l'utilisateur via /auth/signup
        typer.echo("üîê Cr√©ation de l'utilisateur...")
        token = auth.create_user(name=name, email=email, password=password)

        if token:
            typer.echo("‚úÖ Utilisateur cr√©√© avec succ√®s!")
            typer.echo(f"   üë§ Nom: {name}")
            typer.echo(f"   üìß Email: {email}")
            typer.echo()
            typer.echo("üîë Token d'authentification:")
            typer.echo(f"   {token}")
            typer.echo()
            typer.echo("üìù Token tronqu√© (20 premiers caract√®res):")
            typer.echo(f"   {token[:20]}...")
            typer.echo()

            # Sauvegarder le token dans les variables d'environnement de la session

            if save_token_to_env(token):
                typer.echo(
                    "üíæ Token sauvegard√© dans la session courante et dans .env.local"
                )
            else:
                typer.echo("‚ö†Ô∏è Impossible de sauvegarder le token")
            typer.echo()

            # R√©cup√©rer les informations compl√®tes de l'utilisateur
            typer.echo("üìã R√©cup√©ration des informations utilisateur...")
            user_info = auth.get_logged_user_info(token)

            if user_info:
                typer.echo("‚úÖ Informations utilisateur r√©cup√©r√©es:")
                typer.echo(f"   üÜî ID: {user_info.get('id')}")
                typer.echo(f"   üë§ Nom: {user_info.get('name')}")
                typer.echo(f"   üìß Email: {user_info.get('email')}")

                # Formater la date de cr√©ation de mani√®re plus lisible
                created_at = user_info.get("created_at")
                if created_at:
                    if isinstance(created_at, (int, float)):
                        # Si c'est un timestamp, le convertir
                        from utils.date_utils import parse_unix_timestamp

                        formatted_date = parse_unix_timestamp(created_at).strftime(
                            "%d/%m/%Y √† %H:%M:%S"
                        )
                    else:
                        # Si c'est d√©j√† un objet datetime ou une cha√Æne
                        formatted_date = str(created_at)
                    typer.echo(f"   üìÖ Cr√©√© le: {formatted_date}")
                else:
                    typer.echo("   üìÖ Cr√©√© le: N/A")

                typer.echo()
                typer.echo("‚ú® Utilisateur pr√™t √† utiliser!")
                typer.echo(
                    "üí° Vous pouvez maintenant utiliser la commande 'create' pour cr√©er des VMs"
                )
            else:
                typer.echo(
                    "‚ö†Ô∏è Utilisateur cr√©√© mais impossible de r√©cup√©rer les informations"
                )
        else:
            typer.echo("‚ùå √âchec de la cr√©ation de l'utilisateur")
            raise typer.Exit(1)

    except Exception as e:
        logger.error("Erreur lors de la cr√©ation de l'utilisateur", error=str(e))
        typer.echo(f"‚ùå Erreur lors de la cr√©ation: {e}")
        raise typer.Exit(1)

def create(
    name: str = typer.Option("VM de Jean", "--name", "-n", help="Nom de la VM"),
    email: str = typer.Option(
        "jean@dupont21.com", "--email", "-e", help="Email de l'utilisateur existant"
    ),
    password: str = typer.Option(
        "password123", "--password", "-p", help="Mot de passe de l'utilisateur"
    ),
    use_saved_token: bool = typer.Option(
        False, "--use-token", "-t", help="Utiliser le token sauvegard√© dans la session"
    ),
    os: str = typer.Option("Ubuntu 22.04", "--os", "-o", help="Syst√®me d'exploitation"),
    cores: int = typer.Option(
        2, "--cores", "-c", help="Nombre de c≈ìurs CPU", min=1, max=16
    ),
    ram: int = typer.Option(4, "--ram", "-r", help="RAM en GB", min=1, max=128),
    disk: int = typer.Option(50, "--disk", "-d", help="Disque en GB", min=10, max=2048),
    status: str = typer.Option(
        "stopped", "--status", "-s", help="Statut initial de la VM"
    ),
    verbose: bool = typer.Option(False, "--verbose", "-v", help="Mode verbeux"),
) -> None:
    """
    üñ•Ô∏è Cr√©er une VM pour un utilisateur existant

    Authentifie un utilisateur existant et cr√©e une VM pour lui.
    Peut utiliser un token sauvegard√© ou des identifiants email/mot de passe.

    Exemples:

    \b
    python main.py create
    python main.py create --name "Ma VM" --email "alice@example.com" --password "motdepasse"
    python main.py create -n "VM Test" --ram 8 --disk 100 --verbose
    python main.py create --name "Ma VM" --use-token
    """
    # V√©rifier si on doit utiliser le token sauvegard√©
    if use_saved_token:
        from utils.password_utils import get_token_from_config

        saved_token = get_token_from_config()
        if saved_token:
            typer.echo("üîë Utilisation du token sauvegard√© dans la session")
            # Cr√©er un client API avec le token sauvegard√©
            from utils.api import ApiClient

            api_client = ApiClient(token=saved_token)

            # V√©rifier que le token est valide en r√©cup√©rant les infos utilisateur
            try:
                user_info = api_client.get_user_info()
                typer.echo(
                    f"‚úÖ Token valide pour: {user_info.get('name')} ({user_info.get('email')})"
                )

                # Cr√©er la VM directement avec l'API
                vm_result = api_client.vms.create(
                    user_id=user_info["id"],
                    name=name,
                    operating_system=os,
                    cpu_cores=cores,
                    ram_gb=ram,
                    disk_gb=disk,
                    status=status,
                )

                if vm_result:
                    typer.echo("üéâ VM cr√©√©e avec succ√®s!")
                    typer.echo(f"   üÜî ID: {vm_result.get('id')}")
                    typer.echo(f"   üìù Nom: {vm_result.get('name')}")
                    typer.echo(f"   üíª OS: {vm_result.get('operating_system')}")
                    typer.echo(f"   üîß CPU: {vm_result.get('cpu_cores')} cores")
                    typer.echo(f"   üíæ RAM: {vm_result.get('ram_gb')} GB")
                    typer.echo(f"   üíø Disque: {vm_result.get('disk_gb')} GB")
                    typer.echo(f"   ‚ö° Statut: {vm_result.get('status')}")
                    typer.echo("‚ú® Termin√©!")
                else:
                    typer.echo("‚ùå √âchec de la cr√©ation de la VM")
                    raise typer.Exit(1)

            except Exception as e:
                typer.echo(f"‚ùå Token invalide ou expir√©: {e}")
                typer.echo(
                    "üí° Utilisez --password pour vous authentifier avec email/mot de passe"
                )
                raise typer.Exit(1)
        else:
            typer.echo("‚ùå Aucun token sauvegard√© trouv√©")
            typer.echo(
                "üí° Cr√©ez d'abord un utilisateur avec 'signup' ou utilisez --password"
            )
            raise typer.Exit(1)
    else:
        # Appeler directement la fonction avec le mot de passe
        create_vm(name, email, password, os, cores, ram, disk, status, verbose)

@app.command()
def generate(
    user_count: int = typer.Option(
        50, "--users", "-u", help="Nombre d'utilisateurs √† g√©n√©rer", min=1, max=1000
    ),
    min_vms: int = typer.Option(
        0, "--min-vms", help="Nombre minimum de VMs par utilisateur", min=0, max=10
    ),
    max_vms: int = typer.Option(
        5, "--max-vms", help="Nombre maximum de VMs par utilisateur", min=0, max=20
    ),
    output_file: str = typer.Option(
        "vm_users.json", "--output", "-o", help="Fichier de sortie JSON"
    ),
    verbose: bool = typer.Option(False, "--verbose", "-v", help="Mode verbeux"),
) -> None:
    """
    üé≤ G√©n√©rer des donn√©es factices avec Faker

    G√©n√®re un dataset complet d'utilisateurs fran√ßais avec des VMs r√©alistes.
    Les donn√©es sont sauvegard√©es dans un fichier JSON.

    Exemples:

    \b
    python main.py generate
    python main.py generate --users 100 --max-vms 3
    python main.py generate -u 25 -o mon_dataset.json --verbose
    """
    if min_vms > max_vms:
        typer.echo("‚ùå Le nombre minimum de VMs ne peut pas √™tre sup√©rieur au maximum")
        raise typer.Exit(1)

    typer.echo(
        f"üé≤ G√©n√©ration de {user_count} utilisateurs avec {min_vms}-{max_vms} VMs chacun..."
    )

    try:
        # G√©n√©rer les donn√©es
        users_data = DataGenerator.generate_users_with_vms(
            user_count=user_count, vm_per_user_range=(min_vms, max_vms)
        )

        # Sauvegarder dans le fichier JSON
        output_path = Path(output_file)
        with open(output_path, "w", encoding="utf-8") as f:
            json.dump(users_data, f, indent=4, ensure_ascii=False, default=str)

        # Statistiques
        total_vms = sum(len(user["vms"]) for user in users_data)
        users_with_vms_count = len([u for u in users_data if u["vms"]])

        typer.echo("‚úÖ Donn√©es g√©n√©r√©es avec succ√®s !")
        typer.echo("üìä Statistiques:")
        typer.echo(f"   ‚Ä¢ Utilisateurs: {len(users_data)}")
        typer.echo(f"   ‚Ä¢ VMs totales: {total_vms}")
        typer.echo(f"   ‚Ä¢ Utilisateurs avec VMs: {users_with_vms_count}")
        typer.echo(f"   ‚Ä¢ Moyenne VMs/utilisateur: {total_vms / len(users_data):.1f}")
        typer.echo(f"üìÅ Fichier sauvegard√©: {output_path.absolute()}")

        if verbose:
            typer.echo("\nüîç Aper√ßu des donn√©es g√©n√©r√©es:")
            for i, user in enumerate(users_data[:3]):
                typer.echo(
                    f"   {i + 1}. {user['name']} ({user['email']}) - {len(user['vms'])} VMs"
                )
            if len(users_data) > 3:
                typer.echo(f"   ... et {len(users_data) - 3} autres utilisateurs")

    except Exception as e:
        logger.error("Erreur lors de la g√©n√©ration des donn√©es", error=str(e))
        typer.echo(f"‚ùå Erreur lors de la g√©n√©ration: {e}")
        raise typer.Exit(1)
